###冒泡排序
    var arr = [5,9,12,8,1,2,3,4,66,5,6,45,234,43]
	for(var i =0;i<arr.length-1;i++){
    	for(var j=0;j<arr.length-1-i;j++){
          if(arr[j]>arr[j+1]){
            var temp = arr[j+1]
            arr[j+1] = arr[j]
            arr[j] = temp
         }
       }}
